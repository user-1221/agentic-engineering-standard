Usage

# Set your token
export AES_REGISTRY_KEY=aes_tok_...

# Set the registry URL
export AES_REGISTRY_URL=https://registry.aes-official.com

# Publish a skill
aes publish --skill deploy --registry -o dist/

# Publish a template
aes publish --template --registry -o dist/
